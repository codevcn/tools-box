# List all installed packages globally
npm list -g --depth=0

# Code python open local environment pannel (for user)
subprocess.run(["rundll32.exe", "sysdm.cpl,EditEnvironmentVariables"], shell=True)

# Command to find processes using a specific port (e.g., port 3000)
netstat -ano | findstr :4000

# Find the process by its PID (replace <process_name> with the actual name)
tasklist /FI "PID eq 12345"

# Kill a process by its PID (replace <PID> with the actual process ID), forcefully
taskkill /PID <PID> /F

# Kill a process by its name (replace <process_name> with the actual name), forcefully
# Example: taskkill /IM chrome.exe /F
taskkill /IM <process_name> /F

# Open new powershell tab in windows terminal
wt -w 0 nt -p "Win PowerShell 1" ; wt -w 0 nt -p "Win PowerShell 2"

# Open new cmd tab in windows terminal (then run new command)
start cmd /k "cd express-server && dev.cmd" (if previous command is successful then run next command)
npm run dev
## Or
start cmd /k "cd express-server & dev.cmd" (regardless of previous command success then run next command)
npm run dev

# Git remove item from tracking but keep it locally
# Example: git rm -r --cached node_modules, git rm --cached .env
git rm -r --cached <file_or_folder_name>

# Compare files by hashing (replace <file1> and <file2> with actual file paths)
# Example: certutil -hashfile C:\path\to\file1.txt SHA256, certutil -hashfile C:\path\to\file2.txt SHA256
certutil -hashfile <file1> SHA256
certutil -hashfile <file2> SHA256
